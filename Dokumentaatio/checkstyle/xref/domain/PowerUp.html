6   package domain;
7   
8   import java.io.IOException;
9   import java.io.InputStream;
10  import javax.imageio.ImageIO;
11  import javax.swing.ImageIcon;
12  
13  /**
14   * PowerUp luokka.
15   *
16   * @author Jani
17   */
18  public final class PowerUp extends Item implements Drawable {
19  
20      /**
21       * Luodaan maila ja ladataan sen kuva.
22       *
23       * @param startx powerupin aloitus x-koordinaatti
24       * @param starty powerupin aloitus y-koordinaatti
25       * @throws java.io.IOException jos kuvaa ei l√∂ydy.
26       */
27      public PowerUp(int startx, int starty) throws IOException {
28          x = startx;
29          y = starty;
30          InputStream is = getClass().getClassLoader().getResourceAsStream("powerup_red.png");
31          ImageIcon powerup = new ImageIcon(ImageIO.read(is));
32          image = powerup.getImage();
33  
34          width = image.getWidth(null);
35          height = image.getHeight(null);
36      }
37  
38      /**
39       * Liikuttaa poweruppia yhden pikselin alas.
40       */
41      public void move() {
42          y += 1;
43      }
44  
45  }
